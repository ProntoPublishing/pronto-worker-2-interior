% Paragraph formatting

\setlength{\parskip}{0pt}

% ---- H&J configuration (Interior Standard v1 draft defaults) ----
% Values live in this one block so the Standard can re-bind them.

% Widow and orphan control: hard forbid.
\widowpenalty=10000
\clubpenalty=10000
% A page break right after a hyphenated line reads badly.
\brokenpenalty=5000

% Hyphenation: discourage but allow; forbid ladders.
\hyphenpenalty=1000
\exhyphenpenalty=1000
\doublehyphendemerits=10000
\finalhyphendemerits=5000

% Justification escape valve (see fiction template note).
\emergencystretch=1.5em

% Vertical policy: ragged bottom (see fiction template note).
\raggedbottom

% ============================================================================
% HEADERS AND FOOTERS
% ============================================================================

% Body pagestyle — self-contained named style; see fiction template
% for the fancyhdr plain-redefinition leak this design defends against
% (the root cause of headerless production interiors).
%
% Running headers (Interior Standard v1 draft convention):
% verso = book title, recto = current chapter title.
\fancypagestyle{prontobody}{%
  \fancyhf{}%
  \fancyhead[LE]{\small\itshape {{BOOK_TITLE}}}%
  \fancyhead[RO]{\small\itshape\rightmark}%
  \fancyfoot[LE,RO]{\thepage}%
  \renewcommand{\headrulewidth}{0pt}%
}

% Plain style for chapter openers and front matter (folio only)
\fancypagestyle{plain}{
    \fancyhf{}
    \fancyfoot[LE,RO]{\thepage}
    \renewcommand{\headrulewidth}{0pt}
}

\renewcommand{\chaptermark}[1]{\markright{#1}}
\pagestyle{prontobody}

% ============================================================================
% CHAPTER FORMATTING
% ============================================================================

% Chapter title formatting (more structured for non-fiction)
\titleformat{\chapter}[display]
    {\normalfont\huge\bfseries}
    {\chaptertitlename\ \thechapter}
    {20pt}
    {\Huge}

\titlespacing*{\chapter}{0pt}{50pt}{40pt}

% Section formatting
\titleformat{\section}
    {\normalfont\Large\bfseries}
    {\thesection}
    {1em}
    {}

\titleformat{\subsection}
    {\normalfont\large\bfseries}
    {\thesubsection}
    {1em}
    {}
