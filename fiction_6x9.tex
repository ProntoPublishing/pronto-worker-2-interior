% ============================================================================

% ============================================================================

% Interior Standard v1 §1, 6"x9" row: inner 0.85 [BOUND], outer 0.65
% [TUNE], top 0.75 [TUNE], bottom 0.85 [TUNE] (folio clearance included).
\geometry{
    paperwidth=6in,
    paperheight=9in,
    top=0.75in,
    bottom=0.85in,
    inner=0.85in,

% ============================================================================
% FONTS

% ============================================================================
% TYPOGRAPHY
% ============================================================================

% Interior Standard v1 §2 [BOUND]: 11pt / 14.5pt leading for 6"x9".
% book class 11pt gives 13.6pt baselineskip; 14.5/13.6 = 1.066.
\setstretch{1.066}

% Paragraph formatting — Standard §2 [BOUND]: 1em first-line indent,
% no inter-paragraph space.
\setlength{\parindent}{1em}
\setlength{\parskip}{0pt}

% ---- H&J configuration (Interior Standard v1 draft defaults) ----
% Values live in this one block so the Standard can re-bind them.

% Widow and orphan control: hard forbid.
\widowpenalty=10000
\clubpenalty=10000
% A page break right after a hyphenated line reads badly.
\brokenpenalty=5000

% Hyphenation: discourage but allow; forbid ladders.
\hyphenpenalty=1000
\exhyphenpenalty=1000
\doublehyphendemerits=10000
\finalhyphendemerits=5000

% Justification escape valve: with widow/club at 10000 and a 3.75in
% (fiction) measure, TeX needs somewhere to put the slack or it emits
% overfull lines instead.
\emergencystretch=1.5em

% Vertical policy: ragged bottom. Hard widow/orphan penalties make
% flush-bottom spreads stretch interline space unpredictably (the
% "Underfull \vbox" noise in corpus renders); rag the bottom until the
% Standard rules otherwise.
\raggedbottom

% ============================================================================
% HEADERS AND FOOTERS
% ============================================================================

% Body pagestyle — a SELF-CONTAINED named style, deliberately not the
% global "fancy" defaults: in the shipped fancyhdr, invoking a
% \fancypagestyle-defined style (like the plain redefinition below, hit
% by every front-matter page and chapter opener) leaks its \fancyhf{}
% into the global config, silently wiping globally-configured headers.
% That leak is why production interiors never showed running headers.
% A named style's body re-executes on every \pagestyle switch, so it
% cannot stay wiped.
%
% Running headers (Interior Standard v1 draft convention):
% verso = book title, recto = current chapter title.
% \rightmark is set by \chaptermark for numbered chapters; the
% converter emits explicit \markright{...} after starred chapters and
% clears it on part pages.
\fancypagestyle{prontobody}{%
  \fancyhf{}%
  % Standard §4 [BOUND]: headers in small caps; folio bottom center (§5).
  \fancyhead[LE]{\small\scshape {{BOOK_TITLE}}}%
  \fancyhead[RO]{\small\scshape\rightmark}%
  \fancyfoot[C]{\thepage}%
  \renewcommand{\headrulewidth}{0pt}%
}

% Plain style for chapter openers and front matter (folio only)
\fancypagestyle{plain}{
    \fancyhf{}
    \fancyfoot[C]{\thepage}
    \renewcommand{\headrulewidth}{0pt}
}

\renewcommand{\chaptermark}[1]{\markright{#1}}
\pagestyle{prontobody}

% ============================================================================
% CHAPTER FORMATTING
% ============================================================================

% Chapter opener — Standard §4 [BOUND]: label line in spaced small
% caps; title beneath in display size. Sink ~1/3 page [TUNE].
% \prontolabel is also used by the converter for label-shaped starred
% chapters ("Chapter XXVII", "LETTER IV." — style preserved).
\newcommand{\prontolabel}[1]{\textls[120]{\scshape #1}}

\titleformat{\chapter}[display]
    {\normalfont\centering}
    {\Large\prontolabel{\chaptertitlename\ \thechapter}}
    {1em}
    {\Large}

\titlespacing*{\chapter}{0pt}{0.24\textheight}{40pt}

% ============================================================================
% TABLE OF CONTENTS
% ============================================================================

\renewcommand{\contentsname}{Contents}
\setlength{\cftbeforechapskip}{8pt}
% Standard §3.5 [BOUND]: chapter rows get dot leaders; part rows are
% bold with no leader (book-class default, made explicit).
\renewcommand{\cftchapleader}{\cftdotfill{\cftdotsep}}
\renewcommand{\cftpartfont}{\bfseries}
\renewcommand{\cftpartleader}{\hfill}
\renewcommand{\cftpartpagefont}{\bfseries}

% ============================================================================
% CUSTOM COMMANDS
% ============================================================================

% Scene break — Standard §4 [BOUND]: centered asterism, one blank line
% above and below; the FOLLOWING paragraph is flush (§2 [BOUND]).
% \@afterheading suppresses the next paragraph's indent even across
% the blank line the converter emits between blocks.
\makeatletter
\newcommand{\scenebreak}{%
    \par\vspace{\baselineskip}%
    {\centering * * *\par}%
    \vspace{\baselineskip}%
    \@afterindentfalse\@afterheading
}
\makeatother

% ============================================================================
% DOCUMENT
% ============================================================================

\begin{document}

% Front matter
\frontmatter
\pagestyle{plain}

% Half title (bare book title, recto, no folio)
\thispagestyle{empty}
\vspace*{0.28\textheight}
\begin{center}
{\Large {{BOOK_TITLE}}}
\end{center}
\clearpage
% Half-title verso intentionally blank
\thispagestyle{empty}\null\clearpage

% Title page is H-001-conditional. pronto_worker_2.py fills the
% placeholder on the next line: when H-001 fired (author supplied a
% title page; system one suppressed) the placeholder becomes a
% suppression comment. Otherwise it becomes the standard system
% title page block, with the book title and author name substituted
% in the next template-fill step. Do NOT reference the placeholder
% literal in this comment — Python's str.replace() substitutes every
% occurrence and the multi-line replacement breaks out of '%' at its
% first newline. count=1 in the worker is the defense; this comment
% wording is the second layer.
{{SYSTEM_TITLE_PAGE}}

% Copyright page — Standard §3.4 [BOUND] template text.
\thispagestyle{empty}
\vspace*{\fill}
\begin{flushleft}
\small
Copyright \textcopyright\ {{YEAR}} {{AUTHOR_NAME}}. All rights reserved.{{ISBN_LINE}}\\[1em]
Interior design and typesetting by Pronto Publishing\\
prontopublishing.com\\[1em]
First edition, {{YEAR}}
\end{flushleft}
\clearpage

% Table of contents — Standard §3.5 [BOUND]: included when the artifact
% has >= 2 landmark/heading entries. The fill step resolves the
% placeholder to \tableofcontents or an omission comment.
{{TOC_BLOCK}}

% Front matter ends EVEN — Interior Standard v1 §5b accounting:
% \mainmatter resets the page counter, so the end-of-book parity check
% (see END OF BOOK below) sees body pages only. An even front matter
% makes total parity equal body parity, and puts body page 1 on a
% recto under openany.
\clearpage
\ifodd\value{page}\else\thispagestyle{empty}\null\clearpage\fi

% Main matter
\mainmatter
\pagestyle{prontobody}
% Clear running-header marks carried over from the front matter:
% \tableofcontents sets both marks to CONTENTS, and without this clear
% every recto between the TOC and the first chapter's own \markright
% shows a stale "CONTENTS" header (§6 review finding, 2026-07-16).
% Until the first chapter sets its mark, the book title (verso) is the
% only running-header content.
\markboth{}{}

% ============================================================================
% CONTENT PLACEHOLDER
% ============================================================================

{{CONTENT}}

% ============================================================================
% BACK MATTER
% ============================================================================

\backmatter

% Back-matter blocks (About the Author, Also By, etc.) are produced by
% Worker 1 and emitted as part of the body content above. Do not
% reference the template content placeholder literally in comments: a
% stray literal here would be substituted with the full body during
% template fill, rendering the book twice.

% ============================================================================
% END OF BOOK — Interior Standard v1 §5b [BOUND]
% ============================================================================

% The book ends with at least one INTENTIONAL trailing blank page (no
% folio, no header) and an even total page count by construction. We
% control our own output — never rely on the printer's auto-appended
% blank. With the front matter forced even above, body-page parity
% (\value{page} after \mainmatter's reset) equals total parity: after
% the mandatory blank ships, an even counter means an odd total, so one
% more blank ships.
\clearpage
\thispagestyle{empty}\null\clearpage
\ifodd\value{page}\else\thispagestyle{empty}\null\clearpage\fi

\end{document}
