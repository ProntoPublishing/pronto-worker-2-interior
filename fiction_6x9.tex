% ---- H&J configuration (Interior Standard v1 draft defaults) ----
% Values live in this one block so the Standard can re-bind them.

% Widow and orphan control: hard forbid.
\widowpenalty=10000
\clubpenalty=10000
% A page break right after a hyphenated line reads badly.
\brokenpenalty=5000

\raggedbottom

% ============================================================================
% HEADERS AND FOOTERS
% ============================================================================

% Body pagestyle — a SELF-CONTAINED named style, deliberately not the
% global "fancy" defaults: in the shipped fancyhdr, invoking a
% \fancypagestyle-defined style (like the plain redefinition below, hit
% by every front-matter page and chapter opener) leaks its \fancyhf{}
% into the global config, silently wiping globally-configured headers.
% That leak is why production interiors never showed running headers.
% A named style's body re-executes on every \pagestyle switch, so it
% cannot stay wiped.
%
% Running headers (Interior Standard v1 draft convention):
% verso = book title, recto = current chapter title.
% \rightmark is set by \chaptermark for numbered chapters; the
% converter emits explicit \markright{...} after starred chapters and
% clears it on part pages.
\fancypagestyle{prontobody}{%
  \fancyhf{}%
  \fancyhead[LE]{\small\itshape {{BOOK_TITLE}}}%
  \fancyhead[RO]{\small\itshape\rightmark}%
  \fancyfoot[C]{\thepage}%
  \renewcommand{\headrulewidth}{0pt}%
}

% Plain style for chapter openers and front matter (folio only)
\fancypagestyle{plain}{
    \fancyhf{}
    \fancyfoot[C]{\thepage}
    \renewcommand{\headrulewidth}{0pt}
}

\renewcommand{\chaptermark}[1]{\markright{#1}}
\pagestyle{prontobody}
