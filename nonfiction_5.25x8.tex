\setstretch{1.066}

% Paragraph formatting
\setlength{\parindent}{1em}  % Standard s2 [BOUND]
\setlength{\parskip}{0pt}

% ---- H&J configuration (Interior Standard v1 draft defaults) ----
% Values live in this one block so the Standard can re-bind them.

% Widow and orphan control: hard forbid.
\widowpenalty=10000
\clubpenalty=10000
% A page break right after a hyphenated line reads badly.
\brokenpenalty=5000

% Hyphenation: discourage but allow; forbid ladders.
\hyphenpenalty=1000
\exhyphenpenalty=1000
\doublehyphendemerits=10000
\finalhyphendemerits=5000

% Justification escape valve (see fiction template note).
\emergencystretch=1.5em

% Vertical policy: ragged bottom (see fiction template note).
\raggedbottom

% ============================================================================
% HEADERS AND FOOTERS
% ============================================================================

% Body pagestyle — self-contained named style; see fiction template
% for the fancyhdr plain-redefinition leak this design defends against
% (the root cause of headerless production interiors).
%
% Running headers (Interior Standard v1 draft convention):
% verso = book title, recto = current chapter title.
\fancypagestyle{prontobody}{%
  \fancyhf{}%
  \fancyhead[LE]{\small\scshape {{BOOK_TITLE}}}%
  \fancyhead[RO]{\small\scshape\rightmark}%
  \fancyfoot[C]{\thepage}%  % Standard s5 [BOUND]: bottom center
  \renewcommand{\headrulewidth}{0pt}%
}

% Plain style for chapter openers and front matter (folio only)
\fancypagestyle{plain}{
    \fancyhf{}
    \fancyfoot[C]{\thepage}
    \renewcommand{\headrulewidth}{0pt}
}

\renewcommand{\chaptermark}[1]{\markright{#1}}
\pagestyle{prontobody}

% ============================================================================
% CHAPTER FORMATTING
% ============================================================================

% Chapter opener — Standard §4 [BOUND]: spaced-small-caps label, title
% beneath in display size. Sink ~1/3 page [TUNE]. Nonfiction keeps its
% bold title weight (a [TUNE] genre distinction; label style is BOUND).
\newcommand{\prontolabel}[1]{\textls[120]{\scshape #1}}

\titleformat{\chapter}[display]
    {\normalfont\centering}
    {\Large\prontolabel{\chaptertitlename\ \thechapter}}
    {1em}
    {\Large\bfseries}

\titlespacing*{\chapter}{0pt}{0.24\textheight}{40pt}

% Section formatting
\titleformat{\section}
    {\normalfont\Large\bfseries}
    {\thesection}
    {1em}
    {}

\titleformat{\subsection}
    {\normalfont\large\bfseries}
    {\thesubsection}
    {1em}
    {}

% ============================================================================
% TABLE OF CONTENTS
% ============================================================================

\renewcommand{\contentsname}{Contents}
\setlength{\cftbeforechapskip}{8pt}

% no leader.
\renewcommand{\cftchapleader}{\cftdotfill{\cftdotsep}}
\renewcommand{\cftpartfont}{\bfseries}
\renewcommand{\cftpartleader}{\hfill}
\renewcommand{\cftpartpagefont}{\bfseries}

% CUSTOM COMMANDS

% Scene break — Standard §4 [BOUND]: centered asterism, blank line
% above/below; following paragraph flush (§2 [BOUND]).
\makeatletter
\newcommand{\scenebreak}{%
    \par\vspace{\baselineskip}%
    {\centering * * *\par}%
    \vspace{\baselineskip}%
    \@afterindentfalse\@afterheading
}
\makeatother

% Front matter ends EVEN — Interior Standard v1 §5b accounting:
% \mainmatter resets the page counter, so the end-of-book parity check
% (see END OF BOOK below) sees body pages only. An even front matter
% makes total parity equal body parity, and puts body page 1 on a
% recto under openany.
\clearpage
\ifodd\value{page}\else\thispagestyle{empty}\null\clearpage\fi

% Main matter
\mainmatter
\pagestyle{prontobody}
% Clear running-header marks carried over from the front matter:
% \tableofcontents sets both marks to CONTENTS, and without this clear
% every recto between the TOC and the first chapter's own \markright
% shows a stale "CONTENTS" header (§6 review finding, 2026-07-16).
% Until the first chapter sets its mark, the book title (verso) is the
% only running-header content.
\markboth{}{}

% CONTENT PLACEHOLDER
% ============================================================================

{{CONTENT}}

% BACK MATTER
% ============================================================================

\backmatter

% Back-matter blocks (About the Author, Index, etc.) are produced by
% Worker 1 and emitted as part of the body content above. Do not
% reference the template content placeholder literally in comments: a
% stray literal here would be substituted with the full body during
% template fill, rendering the book twice.

% ============================================================================
% END OF BOOK — Interior Standard v1 §5b [BOUND]
% ============================================================================

% The book ends with at least one INTENTIONAL trailing blank page (no
% folio, no header) and an even total page count by construction. We
% control our own output — never rely on the printer's auto-appended
% blank. With the front matter forced even above, body-page parity
% (\value{page} after \mainmatter's reset) equals total parity: after
% the mandatory blank ships, an even counter means an odd total, so one
% more blank ships.
\clearpage
\thispagestyle{empty}\null\clearpage
\ifodd\value{page}\else\thispagestyle{empty}\null\clearpage\fi

\end{document}
